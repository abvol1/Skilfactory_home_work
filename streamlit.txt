    def get_vsp_status_for_date(self, filial_id: int, target_date):
        """
        Возвращает DataFrame со списком всех ВСП филиала и их статусом на указанную дату.
        Статусы:
          - 'Выходной'   – если для ВСП есть запись в vsp_non_working_days на эту дату;
          - 'Заполнен'   – если есть завершённая сессия (status_bul = TRUE) на эту дату;
          - 'Не заполнен' – если ни того, ни другого.
        """
        # Все ВСП филиала
        vsp_list = self.get_vsp_by_filial(filial_id)
        if vsp_list.empty:
            return pd.DataFrame(columns=["ВСП", "Статус"])

        # Получаем список ВСП, у которых эта дата – нерабочая
        non_working_df = self._to_df(
            f"SELECT vsp_id FROM {self._table_name('vsp_non_working_days')} "
            f"WHERE date = %s AND filial_id = %s",
            (target_date, filial_id)
        )
        non_working_ids = set(non_working_df['vsp_id'].tolist()) if not non_working_df.empty else set()

        # Получаем список ВСП, по которым есть завершённая сессия на эту дату
        completed_df = self._to_df(
            f"SELECT vsp_id FROM {self._table_name('checklist_sessions')} "
            f"WHERE operation_date = %s AND filial_id = %s AND status_bul = TRUE",
            (target_date, filial_id)
        )
        completed_ids = set(completed_df['vsp_id'].tolist()) if not completed_df.empty else set()

        # Формируем результат
        result = []
        for _, vsp_row in vsp_list.iterrows():
            vid = vsp_row['id']
            vname = vsp_row['name']
            if vid in non_working_ids:
                status = "Выходной"
            elif vid in completed_ids:
                status = "Заполнен"
            else:
                status = "Не заполнен"
            result.append({"ВСП": vname, "Статус": status})

        return pd.DataFrame(result)

if tab_visualization is not None:
    with tab_visualization:
        st.markdown("## 📊 Статус заполнения по ВСП")
        st.caption("Выберите филиал и дату, чтобы увидеть, какие ВСП уже отчитались, а какие — нет.")

        col1, col2 = st.columns(2)

            # Выбор филиала

                filial_names = filials_df['name'].tolist()
                sel_filial_name = st.selectbox("🏢 Филиал", filial_names, key="vis_filial")
                filial_id_vis = int(filials_df[filials_df['name'] == sel_filial_name]['id'].iloc[0])
            else:
                st.error("Нет филиалов в базе")
                st.stop()
        with col2:

        if st.button("🔍 Показать статус", use_container_width=True):
            with st.spinner("Собираем данные..."):
                status_df = db.get_vsp_status_for_date(filial_id_vis, vis_date)

            if status_df.empty:
                st.info("Нет данных по выбранному филиалу.")
            else:
                # 1. Столбчатая диаграмма
                st.subheader("📈 Распределение статусов")
                status_counts = status_df['Статус'].value_counts()
                # Чтобы столбцы всегда были в одном порядке
                ordered_counts = {}
                for s in ["Заполнен", "Не заполнен", "Выходной"]:
                    if s in status_counts:
                        ordered_counts[s] = status_counts[s]
                if ordered_counts:
                    chart_df = pd.DataFrame({
                        "Статус": list(ordered_counts.keys()),
                        "Количество": list(ordered_counts.values())
                    })
                    st.bar_chart(chart_df.set_index("Статус"))

                # 2. Таблица с цветовой индикацией
                st.subheader("📋 Детализация по ВСП")
                # Функция для окрашивания строк
                def highlight_status(row):
                    if row['Статус'] == 'Заполнен':
                        return ['background-color: #d4edda; color: #155724'] * len(row)  # зеленый
                    elif row['Статус'] == 'Выходной':
                        return ['background-color: #ffeeba; color: #856404'] * len(row)  # желтый
                    else:  # Не заполнен
                        return ['background-color: #f8d7da; color: #721c24'] * len(row)  # красный

                styled_df = status_df.style.apply(highlight_status, axis=1)
                st.dataframe(styled_df, use_container_width=True, height=400)
5. Пояснение работы
На вкладке администратор выбирает филиал и дату.

При нажатии кнопки «Показать статус» для каждого ВСП филиала определяется статус:

Выходной – в таблице нерабочих дней есть запись на эту дату.

Заполнен – есть завершённая сессия (status_bul = TRUE) с operation_date = выбранная дата.

Не заполнен – нет ни того, ни другого.

Выводится столбчатая диаграмма с количеством ВСП по статусам и цветная таблица (зелёный/жёлтый/красный).

Таблица показывает только ВСП выбранного филиала.


def session_exists_for_vsp_date(self, vsp_id, operation_date):
    """Проверяет, есть ли уже любая сессия для данного ВСП на указанную дату."""
    row = self._execute(
        f"SELECT id FROM {self._table_name('checklist_sessions')} "
        f"WHERE vsp_id = %s AND operation_date = %s LIMIT 1",
        (vsp_id, operation_date),
        fetch_one=True
    )
    return row is not None





with st.form("new_session_form"):
    op_date = st.date_input("📅 Дата", value=datetime.date.today(), format="DD.MM.YYYY", disabled=True)
    submitted = st.form_submit_button(
        "▶️ НАЧАТЬ ЗАПОЛНЕНИЕ",
        type="primary",
        use_container_width=True,
        disabled=(not new_session_allowed)  # отключаем кнопку, если у пользователя уже есть черновик за сегодня
    )
    if submitted and sel_vsp_id is not None:
        # Проверяем, нет ли уже ЛЮБОЙ сессии (черновик/завершённая) для этого ВСП на сегодня
        if db.session_exists_for_vsp_date(sel_vsp_id, op_date):
            st.error(
                f"❌ Для ВСП «{sel_vsp}» на {op_date} уже существует проверка (черновик или завершённая). "
                "Повторное создание невозможно."
            )
        else:
            # Дополнительная страховка: проверим, что у этого пользователя нет незавершённого черновика за сегодня (для любого ВСП)
            draft_id = db.get_today_draft_session_id(st.session_state.user_full_name)
            if draft_id is not None:
                st.error("У вас уже есть незавершённый черновик за сегодня. Завершите или удалите его.")
            else:
                sid = db.create_session(
                    st.session_state.user_full_name,
                    sel_filial_id,
                    sel_vsp_id,
                    op_date,
                    False
                )
                st.session_state.current_session_id = sid
                st.session_state.step = 1
                st.rerun()
    elif submitted:
        st.error("Выберите ВСП!")

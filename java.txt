import org.apache.poi.ss.usermodel.*;
import org.apache.poi.xssf.usermodel.XSSFWorkbook;

public class CzoProcessor {

    // Индексы столбцов исходного файла CZO (0-based)
    private static final int COL_FILIAL = 1;             // B - Филиал (первые 3 цифры)
    private static final int COL_CLIENT_NAME = 2;        // C - Клиент (ФИО)
    private static final int COL_ACCOUNT_RB = 4;         // E - Счет РБ (лицевой счет)
    private static final int COL_CURRENCY = 5;           // F - Валюта счета
    private static final int COL_BALANCE_RB = 6;         // G - Остаток счета РБ на дату резервирования
    private static final int COL_OVERDUE_BALANCE = 7;    // H - Остаток просроченной РБ на дату резервирования
    private static final int COL_RESERVE_ACCOUNT = 11;   // L - Счет резерва
    private static final int COL_RESERVE_BALANCE = 13;   // N - Остаток счета резерва на дату резервирования
    private static final int COL_CALC_RESERVE = 14;      // O - Расчетный резерв
    private static final int COL_DISCREPANCY = 15;       // P - Расхождение

    // Шаблоны счетов
    private static final Pattern ACCOUNT_60323 = Pattern.compile("^60323.*");
    private static final Pattern ACCOUNT_458 = Pattern.compile("^458.*");
    private static final Pattern ACCOUNT_47423 = Pattern.compile("^47423.*");

    // Строки начала данных в шаблоне
    private static final int RVP_START_ROW = 17;  // 18-я строка (индекс 17)
    private static final int POT_START_ROW = 6;   // 7-я строка (индекс 6)

    /**
     * Главный метод обработки.
     */
    public void process(File inputFile) throws IOException {
        // 1. Чтение и группировка исходных данных
        Map<String, List<RowData>> groupedByFilial = readAndGroupByFilial(inputFile);

        // 2. Для каждого филиала создаём файл на основе шаблона
        for (Map.Entry<String, List<RowData>> entry : groupedByFilial.entrySet()) {
            String filialCode = entry.getKey();
            List<RowData> rows = entry.getValue();
            createOrderFromTemplate(inputFile.getParentFile(), filialCode, rows);
        }
    }

    /**
     * Чтение исходного Excel, группировка по первым трём цифрам филиала.
     */
    private Map<String, List<RowData>> readAndGroupByFilial(File file) throws IOException {
        Map<String, List<RowData>> groups = new HashMap<>();

        try (FileInputStream fis = new FileInputStream(file);
             Workbook workbook = WorkbookFactory.create(fis)) {

            Sheet sheet = workbook.getSheetAt(0);
            DataFormatter formatter = new DataFormatter();

            // Пропускаем заголовок (предполагаем, что первая строка - шапка)
            for (int rowIdx = 1; rowIdx <= sheet.getLastRowNum(); rowIdx++) {
                Row row = sheet.getRow(rowIdx);
                if (row == null) continue;

                // Филиал (первые 3 символа столбца B)
                Cell filialCell = row.getCell(COL_FILIAL);
                String filialRaw = formatter.formatCellValue(filialCell).trim();
                if (filialRaw.isEmpty()) continue;

                String filialKey = filialRaw.length() >= 3 ? filialRaw.substring(0, 3) : filialRaw;
                if (!isTargetFilial(filialKey)) continue;

                RowData data = new RowData();
                data.accountRB = formatter.formatCellValue(row.getCell(COL_ACCOUNT_RB));
                data.clientName = formatter.formatCellValue(row.getCell(COL_CLIENT_NAME));
                data.currency = formatter.formatCellValue(row.getCell(COL_CURRENCY));
                data.balanceRB = getNumericValue(row.getCell(COL_BALANCE_RB));
                data.overdueBalance = getNumericValue(row.getCell(COL_OVERDUE_BALANCE));
                data.reserveAccount = formatter.formatCellValue(row.getCell(COL_RESERVE_ACCOUNT));
                data.reserveBalance = getNumericValue(row.getCell(COL_RESERVE_BALANCE));
                data.calcReserve = getNumericValue(row.getCell(COL_CALC_RESERVE));
                data.discrepancy = getNumericValue(row.getCell(COL_DISCREPANCY));

                groups.computeIfAbsent(filialKey, k -> new ArrayList<>()).add(data);
            }
        }
        return groups;
    }

    private boolean isTargetFilial(String code) {
        try {
            int num = Integer.parseInt(code);
            return (num >= 1 && num <= 3) || (num >= 5 && num <= 9);
        } catch (NumberFormatException e) {
            return false;
        }
    }

    private BigDecimal getNumericValue(Cell cell) {
        if (cell == null) return BigDecimal.ZERO;
        try {
            if (cell.getCellType() == CellType.NUMERIC) {
                return BigDecimal.valueOf(cell.getNumericCellValue());
            } else if (cell.getCellType() == CellType.STRING) {
                String str = cell.getStringCellValue().trim().replace(",", ".");
                if (str.isEmpty()) return BigDecimal.ZERO;
                return new BigDecimal(str);
            }
        } catch (Exception ignored) {}
        return BigDecimal.ZERO;
    }

    /**
     * Создание файла распоряжения из шаблона template.xlsx.
     */
    private void createOrderFromTemplate(File parentDir, String filialCode, List<RowData> allRows) throws IOException {
        // Загружаем шаблон из ресурсов
        try (InputStream templateStream = getClass().getResourceAsStream("/template.xlsx")) {
            if (templateStream == null) {
                throw new IOException("Шаблон template.xlsx не найден в ресурсах!");
            }

            try (Workbook workbook = new XSSFWorkbook(templateStream)) {
                // Фильтрация строк для двух листов
                List<RowData> rvpRows = new ArrayList<>();
                List<RowData> potRows = new ArrayList<>();

                for (RowData row : allRows) {
                    // РВП: счета 60323*,458*,47423* и расхождение != 0
                    if (isRvpAccount(row.accountRB) && row.discrepancy.compareTo(BigDecimal.ZERO) != 0) {
                        rvpRows.add(row);
                    }

                    // ПОТ: счета 60323* (остаток > 0) или 458*/47423* (остаток РБ или просроченный > 0)
                    if (isPotAccount(row.accountRB)) {
                        if (ACCOUNT_60323.matcher(row.accountRB).matches()) {
                            if (row.balanceRB.compareTo(BigDecimal.ZERO) > 0) {
                                potRows.add(row);
                            }
                        } else { // 458* или 47423*
                            if (row.balanceRB.compareTo(BigDecimal.ZERO) > 0 ||
                                row.overdueBalance.compareTo(BigDecimal.ZERO) > 0) {
                                potRows.add(row);
                            }
                        }
                    }
                }

                // Заполняем листы
                Sheet sheetRvp = workbook.getSheet("Распоряжение (РВП)");
                Sheet sheetPot = workbook.getSheet("Расчетная база ПОТ");
                if (sheetRvp == null || sheetPot == null) {
                    throw new IOException("В шаблоне отсутствуют нужные листы!");
                }

                // Заполняем с сохранением форматирования
                fillSheetWithInsert(sheetRvp, rvpRows, RVP_START_ROW, true);
                fillSheetWithInsert(sheetPot, potRows, POT_START_ROW, false);

                // Сохраняем результат
                String outputFileName = String.format("Распоряжение_филиал_%s.xlsx", filialCode);
                File outputFile = new File(parentDir, outputFileName);
                try (FileOutputStream fos = new FileOutputStream(outputFile)) {
                    workbook.write(fos);
                }
            }
        }
    }

    private boolean isRvpAccount(String account) {
        return ACCOUNT_60323.matcher(account).matches() ||
               ACCOUNT_458.matcher(account).matches() ||
               ACCOUNT_47423.matcher(account).matches();
    }

    private boolean isPotAccount(String account) {
        return isRvpAccount(account);
    }

    /**
     * Заполнение листа с динамической вставкой строк и сохранением форматирования.
     * @param sheet лист
     * @param rows данные для заполнения
     * @param startRowIndex индекс строки, с которой начинаются данные (0-based)
     * @param isRvp true для листа "Распоряжение (РВП)", false для "Расчетная база ПОТ"
     */
    private void fillSheetWithInsert(Sheet sheet, List<RowData> rows, int startRowIndex, boolean isRvp) {
        // 1. Найти строку с текстом под таблицей (ищем "Итого" или "Примечание")
        int textStartRow = findTextStartRow(sheet, startRowIndex);
        if (textStartRow == -1) {
            // Если не нашли, считаем что текст начинается через 1 строку после образца
            textStartRow = startRowIndex + 1;
        }

        // 2. Сохраняем строку-образец (она должна быть в шаблоне на позиции startRowIndex)
        Row templateRow = sheet.getRow(startRowIndex);
        if (templateRow == null) {
            templateRow = sheet.createRow(startRowIndex);
        }

        // 3. Определяем сколько строк нужно для данных
        int rowsNeeded = Math.max(rows.size(), 1);

        // 4. Если нужно больше одной строки, вставляем дополнительные
        if (rowsNeeded > 1) {
            // Сдвигаем весь текст вниз, сохраняя форматирование
            // Сдвигаем строки от textStartRow до конца листа на (rowsNeeded - 1) вниз
            int shiftAmount = rowsNeeded - 1;
            if (textStartRow <= sheet.getLastRowNum()) {
                sheet.shiftRows(textStartRow, sheet.getLastRowNum(), shiftAmount, true, false);
            }

            // Копируем стили из строки-образца на новые строки
            for (int i = 1; i < rowsNeeded; i++) {
                Row newRow = sheet.getRow(startRowIndex + i);
                if (newRow == null) {
                    newRow = sheet.createRow(startRowIndex + i);
                }
                copyRowStyleAndBorders(templateRow, newRow);
            }
        }

        // 5. Заполняем данные
        int rowNum = startRowIndex;
        for (RowData data : rows) {
            Row row = sheet.getRow(rowNum);
            if (row == null) {
                row = sheet.createRow(rowNum);
                copyRowStyleAndBorders(templateRow, row);
            }

            if (isRvp) {
                fillRvpRow(row, data);
            } else {
                fillPotRow(row, data);
            }
            rowNum++;
        }

        // 6. Если данных нет - очищаем строку-образец
        if (rows.isEmpty()) {
            Row emptyRow = sheet.getRow(startRowIndex);
            if (emptyRow != null) {
                for (int i = 0; i < emptyRow.getLastCellNum(); i++) {
                    Cell cell = emptyRow.getCell(i);
                    if (cell != null) {
                        cell.setCellValue("");
                    }
                }
            }
        }
    }

    /**
     * Поиск строки, где начинается текст под таблицей (ищем "Итого" или "Примечание" в первом столбце).
     */
    private int findTextStartRow(Sheet sheet, int startFromRow) {
        DataFormatter formatter = new DataFormatter();
        for (int i = startFromRow + 1; i <= sheet.getLastRowNum(); i++) {
            Row row = sheet.getRow(i);
            if (row == null) continue;
            Cell cell = row.getCell(0);
            String val = formatter.formatCellValue(cell).trim().toLowerCase();
            if (val.contains("итого") || val.contains("примечание") || val.contains("всего")) {
                return i;
            }
        }
        return -1;
    }

    /**
     * Копирование стилей ячеек и границ из одной строки в другую.
     * Это сохраняет все линии таблицы.
     */
    private void copyRowStyleAndBorders(Row source, Row target) {
        // Копируем высоту строки
        target.setHeight(source.getHeight());

        for (int i = 0; i < source.getLastCellNum(); i++) {
            Cell srcCell = source.getCell(i);
            Cell tgtCell = target.createCell(i);

            if (srcCell != null) {
                // Копируем стиль (включает границы, шрифт, выравнивание, формат чисел)
                CellStyle srcStyle = srcCell.getCellStyle();
                CellStyle tgtStyle = target.getSheet().getWorkbook().createCellStyle();
                tgtStyle.cloneStyleFrom(srcStyle);
                tgtCell.setCellStyle(tgtStyle);

                // Копируем тип ячейки (но не значение)
                // Значение будет установлено позже при заполнении
            } else {
                // Если в образце ячейки нет, но в других строках она может быть
                tgtCell.setCellStyle(target.getSheet().getWorkbook().createCellStyle());
            }
        }
    }

    /**
     * Заполнение строки листа "Распоряжение (РВП)".
     * Столбцы согласно ТЗ (вкладка Распоряжение (РВП)):
     * 1. № п/п
     * 2. Номер лицевого счета, на котором отражен элемент расчетной базы (= Счет РБ)
     * 3. Сумма элемента расчетной базы (в рублях) (= Расчетный резерв)
     * 4. Категория качества элемента расчетной базы (= 5)
     * 5. Размер расчетного резерва (в % от суммы) (= 100)
     * 6. Размер расчетного резерва (в рублях) (= Расчетный резерв)
     * 7. Минимальный размер резерва (в рублях) (= Расчетный резерв)
     * 8. Формируемый резерв на возможные потери (в рублях) (= Расчетный резерв)
     * 9. Доля формируемого резерва в объеме задолженности (= 1)
     * 10. Фактически сформированный резерв на возможные потери (в рублях) (= Остаток счета резерва)
     * 11. Изменение резерва (+,-) (в рублях) (= Расхождение)
     */
    private void fillRvpRow(Row row, RowData data) {
        int col = 0;

        // 1. № п/п - будет заполнен позже (пока оставляем пустым или заполним в цикле)
        // Номер будем проставлять отдельно после заполнения всех строк
        // row.createCell(col++).setCellValue(seq);

        // 2. Номер лицевого счета = Счет РБ
        row.createCell(col++).setCellValue(data.accountRB);

        // 3. Сумма элемента расчетной базы = Расчетный резерв
        setNumericCell(row.createCell(col++), data.calcReserve);

        // 4. Категория качества = 5
        row.createCell(col++).setCellValue(5);

        // 5. Размер расчетного резерва (%) = 100
        row.createCell(col++).setCellValue(100);

        // 6. Размер расчетного резерва (руб) = Расчетный резерв
        setNumericCell(row.createCell(col++), data.calcReserve);

        // 7. Минимальный размер резерва = Расчетный резерв
        setNumericCell(row.createCell(col++), data.calcReserve);

        // 8. Формируемый резерв = Расчетный резерв
        setNumericCell(row.createCell(col++), data.calcReserve);

        // 9. Доля формируемого резерва = 1
        row.createCell(col++).setCellValue(1);

        // 10. Фактически сформированный резерв = Остаток счета резерва
        setNumericCell(row.createCell(col++), data.reserveBalance);

        // 11. Изменение резерва = Расхождение
        setNumericCell(row.createCell(col++), data.discrepancy);
    }

    /**
     * Заполнение строки листа "Расчетная база ПОТ".
     * Столбцы согласно ТЗ:
     * 1. ФИО клиента
     * 2. Номер договора (нет данных - пусто)
     * 3. Дата заключения договора (нет данных - пусто)
     * 4. Признак принадлежности к ПОТ на дату "Дата отчета предыдущего регулирования" (нет данных - пусто)
     * 5. Признак принадлежности к ПОТ на дату "Дата отчета" (нет данных - пусто)
     * 6. Валюта
     * 7. Коэффициент резервирования
     * 8. Категория качества
     * 9. Счет для отчетности (= Счет РБ)
     * 10. Остаток по счету ЭРБ
     * 11. Остаток по счету ЭРБ до 30 дней (резервированию не подлежит)
     */
    private void fillPotRow(Row row, RowData data) {
        int col = 0;

        // 1. ФИО клиента
        row.createCell(col++).setCellValue(data.clientName);

        // 2. Номер договора - пусто
        row.createCell(col++).setCellValue("");

        // 3. Дата заключения договора - пусто
        row.createCell(col++).setCellValue("");

        // 4. Признак принадлежности к ПОТ на предыдущую дату - пусто
        row.createCell(col++).setCellValue("");

        // 5. Признак принадлежности к ПОТ на дату отчета - пусто
        row.createCell(col++).setCellValue("");

        // 6. Валюта
        row.createCell(col++).setCellValue(data.currency);

        // === Расчёт коэффициента резервирования и категории качества ===
        BigDecimal reserveCoef;
        int qualityCategory;

        if (ACCOUNT_60323.matcher(data.accountRB).matches()) {
            // Для счета 60323* - всегда 100% и 5 категория
            reserveCoef = new BigDecimal("100");
            qualityCategory = 5;
        } else {
            // Для счетов 458*, 47423*
            // Остаток по счету ЭРБ = остаток просроченной РБ (столбец H)
            BigDecimal erbBalance = data.overdueBalance;

            if (erbBalance.compareTo(BigDecimal.ZERO) > 0) {
                reserveCoef = new BigDecimal("100");
                qualityCategory = 5;
            } else {
                reserveCoef = BigDecimal.ZERO;
                qualityCategory = 1;
            }
        }

        // 7. Коэффициент резервирования
        setNumericCell(row.createCell(col++), reserveCoef);

        // 8. Категория качества
        row.createCell(col++).setCellValue(qualityCategory);

        // 9. Счет для учета ЭРБ = Счет РБ
        row.createCell(col++).setCellValue(data.accountRB);

        // 10. Остаток по счету ЭРБ
        BigDecimal erbBalanceValue;
        if (ACCOUNT_60323.matcher(data.accountRB).matches()) {
            erbBalanceValue = data.balanceRB;
        } else {
            erbBalanceValue = data.overdueBalance;
        }
        setNumericCell(row.createCell(col++), erbBalanceValue);

        // 11. Остаток по счету ЭРБ до 30 дней = Остаток счета РБ на дату резервирования
        setNumericCell(row.createCell(col++), data.balanceRB);
    }

    /**
     * Проставляет порядковые номера в первом столбце для листа РВП.
     * Вызывается после заполнения всех строк данных.
     */
    private void setSequenceNumbers(Sheet sheet, int startRow, int rowCount) {
        for (int i = 0; i < rowCount; i++) {
            Row row = sheet.getRow(startRow + i);
            if (row != null) {
                Cell seqCell = row.getCell(0);
                if (seqCell == null) {
                    seqCell = row.createCell(0);
                }
                seqCell.setCellValue(i + 1);
            }
        }
    }

    private void setNumericCell(Cell cell, BigDecimal value) {
        if (value == null) value = BigDecimal.ZERO;
        cell.setCellValue(value.doubleValue());
    }

    }

/**
 * Заполнение листа с динамической вставкой строк и сохранением форматирования.
 * @param sheet лист
 * @param rows данные для заполнения
 * @param startRowIndex индекс строки, с которой начинаются данные (0-based)
 * @param isRvp true для листа "Распоряжение (РВП)", false для "Расчетная база ПОТ"
 */
private void fillSheetWithInsert(Sheet sheet, List<RowData> rows, int startRowIndex, boolean isRvp) {
    // 1. Найти строку с текстом под таблицей (ищем "Итого" или "Примечание")
    int textStartRow = findTextStartRow(sheet, startRowIndex);
    if (textStartRow == -1) {
        // Если не нашли, считаем что текст начинается через 1 строку после образца
        textStartRow = startRowIndex + 1;
    }

    // 2. Сохраняем строку-образец (она должна быть в шаблоне на позиции startRowIndex)
    Row templateRow = sheet.getRow(startRowIndex);
    if (templateRow == null) {
        templateRow = sheet.createRow(startRowIndex);
    }

    // 3. Определяем сколько строк нужно для данных
    int rowsNeeded = Math.max(rows.size(), 1);

    // 4. Если нужно больше одной строки, вставляем дополнительные
    if (rowsNeeded > 1) {
        // Сдвигаем весь текст вниз, сохраняя форматирование
        // Сдвигаем строки от textStartRow до конца листа на (rowsNeeded - 1) вниз
        int shiftAmount = rowsNeeded - 1;
        if (textStartRow <= sheet.getLastRowNum()) {
            sheet.shiftRows(textStartRow, sheet.getLastRowNum(), shiftAmount, true, false);
        }

        // Копируем стили из строки-образца на новые строки
        for (int i = 1; i < rowsNeeded; i++) {
            Row newRow = sheet.getRow(startRowIndex + i);
            if (newRow == null) {
                newRow = sheet.createRow(startRowIndex + i);
            }
            copyRowStyleAndBorders(templateRow, newRow);
        }
    }

    // 5. Заполняем данные
    int rowNum = startRowIndex;
    for (RowData data : rows) {
        Row row = sheet.getRow(rowNum);
        if (row == null) {
            row = sheet.createRow(rowNum);
            copyRowStyleAndBorders(templateRow, row);
        }

        if (isRvp) {
            fillRvpRow(row, data);
        } else {
            fillPotRow(row, data);
        }
        rowNum++;
    }

    // 6. Если данных нет - очищаем строку-образец
    if (rows.isEmpty()) {
        Row emptyRow = sheet.getRow(startRowIndex);
        if (emptyRow != null) {
            for (int i = 0; i < emptyRow.getLastCellNum(); i++) {
                Cell cell = emptyRow.getCell(i);
                if (cell != null) {
                    cell.setCellValue("");
                }
            }
        }
    }

    // 7. Для листа РВП проставляем порядковые номера в первом столбце
    if (isRvp && !rows.isEmpty()) {
        for (int i = 0; i < rows.size(); i++) {
            Row row = sheet.getRow(startRowIndex + i);
            if (row != null) {
                Cell seqCell = row.getCell(0);
                if (seqCell == null) {
                    seqCell = row.createCell(0);
                }
                seqCell.setCellValue(i + 1);
            }
        }
    }
}

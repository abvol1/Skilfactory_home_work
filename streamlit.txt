3. DatabaseManager (Java 8 адаптация)

Единственное изменение – убираем var и используем явные типы (но в Java 8 var нет, поэтому код изначально был без него). Остальной код совместим. Убедитесь, что в методе checkUserByLogin обрабатывается wasNull() корректно. Приведу полный класс для уверенности:

```java
package com.example.checklist.database;

import com.example.checklist.models.*;
import java.io.InputStream;
import java.io.IOException;
import java.sql.*;
import java.util.ArrayList;
import java.util.List;
import java.util.Properties;

public class DatabaseManager {
    private static String URL;
    private static String USER;
    private static String PASSWORD;
    private static String SCHEMA;

    static {
        try (InputStream input = DatabaseManager.class.getClassLoader()
                .getResourceAsStream("application.properties")) {
            Properties prop = new Properties();
            if (input != null) {
                prop.load(input);
            }
            String host = prop.getProperty("db.host", "localhost");
            String port = prop.getProperty("db.port", "5432");
            String dbName = prop.getProperty("db.name", "postgres");
            URL = "jdbc:postgresql://" + host + ":" + port + "/" + dbName;
            USER = prop.getProperty("db.user", "postgres");
            PASSWORD = prop.getProperty("db.password", "");
            SCHEMA = prop.getProperty("db.schema", "public");
        } catch (IOException ex) {
            ex.printStackTrace();
            // fallback defaults
            URL = "jdbc:postgresql://localhost:5432/postgres";
            USER = "postgres";
            PASSWORD = "";
            SCHEMA = "public";
        }
    }

    private Connection getConnection() throws SQLException {
        return DriverManager.getConnection(URL, USER, PASSWORD);
    }

    public List<Filial> getFilials() {
        String query = "SELECT id, name, check_name FROM " + SCHEMA + ".filials ORDER BY name";
        List<Filial> filials = new ArrayList<>();
        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(query)) {
            while (rs.next()) {
                Filial f = new Filial();
                f.setId(rs.getInt("id"));
                f.setName(rs.getString("name"));
                f.setCheckName(rs.getBoolean("check_name"));
                filials.add(f);
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return filials;
    }

    public List<Vsp> getVspByFilial(int filialId) {
        String query = "SELECT id, name FROM " + SCHEMA + ".vsp WHERE filial_id = ? ORDER BY name";
        List<Vsp> vspList = new ArrayList<>();
        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(query)) {
            pstmt.setInt(1, filialId);
            ResultSet rs = pstmt.executeQuery();
            while (rs.next()) {
                vspList.add(new Vsp(rs.getInt("id"), rs.getString("name")));
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return vspList;
    }

    public User checkUserByLogin(String login) {
        String query = "SELECT us.full_name, f.name AS filial_name, f.id AS filial_id " +
                       "FROM " + SCHEMA + ".users us " +
                       "LEFT JOIN " + SCHEMA + ".filials f ON us.name_filial::numeric = f.id " +
                       "WHERE LOWER(us.name) = LOWER(?)";
        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(query)) {
            pstmt.setString(1, login.trim());
            ResultSet rs = pstmt.executeQuery();
            if (rs.next()) {
                String fullName = rs.getString("full_name");
                String filialName = rs.getString("filial_name");
                int filialId = rs.getInt("filial_id");
                if (rs.wasNull()) filialId = -1;
                return new User(login, fullName, filialId, filialName);
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return null;
    }

    public List<ChecklistTemplateItem> getChecklistTemplate() {
        String query = "SELECT id, item_order, description, additional_info, filter_value, " +
                       "events_value, alt_filter_value, alt_additional_info, alt_events_value " +
                       "FROM " + SCHEMA + ".checklist_templates ORDER BY item_order";
        List<ChecklistTemplateItem> items = new ArrayList<>();
        try (Connection conn = getConnection();
             Statement stmt = conn.createStatement();
             ResultSet rs = stmt.executeQuery(query)) {
            while (rs.next()) {
                ChecklistTemplateItem item = new ChecklistTemplateItem();
                item.setId(rs.getInt("id"));
                item.setOrder(rs.getInt("item_order"));
                item.setDescription(rs.getString("description"));
                item.setAdditionalInfo(rs.getString("additional_info"));
                item.setFilterValue(rs.getString("filter_value"));
                item.setEventsValue(rs.getString("events_value"));
                item.setAltFilterValue(rs.getString("alt_filter_value"));
                item.setAltAdditionalInfo(rs.getString("alt_additional_info"));
                item.setAltEventsValue(rs.getString("alt_events_value"));
                items.add(item);
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return items;
    }

    public boolean getFilialCheckName(int filialId) {
        String query = "SELECT check_name FROM " + SCHEMA + ".filials WHERE id = ?";
        try (Connection conn = getConnection();
             PreparedStatement pstmt = conn.prepareStatement(query)) {
            pstmt.setInt(1, filialId);
            ResultSet rs = pstmt.executeQuery();
            if (rs.next()) {
                return rs.getBoolean("check_name");
            }
        } catch (SQLException e) {
            e.printStackTrace();
        }
        return false;
    }
}
```

4. MainController (Java 8 адаптация)

В JavaFX 8 Platform.runLater принимает Runnable. Лямбды работают. Единственное – Font.font("System", FontWeight.BOLD, 16) – такой шрифт есть. Всё совместимо. Приведу код с явными типами для уверенности:

```java
package com.example.checklist.controllers;

import com.example.checklist.database.DatabaseManager;
import com.example.checklist.models.*;
import com.example.checklist.utils.ClipboardHelper;
import javafx.application.Platform;
import javafx.fxml.FXML;
import javafx.fxml.Initializable;
import javafx.geometry.Insets;
import javafx.scene.control.*;
import javafx.scene.layout.HBox;
import javafx.scene.layout.VBox;
import javafx.scene.text.Font;
import javafx.scene.text.FontWeight;
import org.controlsfx.control.PopOver;

import java.net.URL;
import java.time.LocalDate;
import java.time.format.DateTimeFormatter;
import java.util.List;
import java.util.ResourceBundle;

public class MainController implements Initializable {
    @FXML private VBox loginPanel;
    @FXML private TextField loginField;
    @FXML private Label loginErrorLabel;
    @FXML private VBox mainPanel;
    @FXML private Label userInfoLabel;
    @FXML private ComboBox<String> vspCombo;
    @FXML private DatePicker datePicker;
    @FXML private ScrollPane checklistScroll;
    @FXML private VBox checklistContainer;

    private DatabaseManager db = new DatabaseManager();
    private User currentUser;
    private List<Vsp> vspList;

    @Override
    public void initialize(URL location, ResourceBundle resources) {
        loginPanel.setVisible(true);
        mainPanel.setVisible(false);
    }

    @FXML
    private void handleLogin() {
        String login = loginField.getText().trim();
        if (login.isEmpty()) {
            loginErrorLabel.setText("Введите логин");
            return;
        }
        User user = db.checkUserByLogin(login);
        if (user == null || user.getFilialId() <= 0) {
            loginErrorLabel.setText("Пользователь не найден или отсутствует филиал");
        } else {
            currentUser = user;
            loginPanel.setVisible(false);
            mainPanel.setVisible(true);
            userInfoLabel.setText("Пользователь: " + user.getFullName() +
                    "  |  Филиал: " + user.getFilialName());
            loadVspList();
        }
    }

    private void loadVspList() {
        vspList = db.getVspByFilial(currentUser.getFilialId());
        vspCombo.getItems().clear();
        for (Vsp v : vspList) {
            vspCombo.getItems().add(v.getName());
        }
        if (!vspList.isEmpty()) {
            vspCombo.getSelectionModel().selectFirst();
        }
    }

    @FXML
    private void handleShowChecklist() {
        if (currentUser == null) return;
        String selectedVspName = vspCombo.getValue();
        if (selectedVspName == null) {
            showAlert("Выберите ВСП");
            return;
        }
        Vsp selectedVsp = null;
        for (Vsp v : vspList) {
            if (v.getName().equals(selectedVspName)) {
                selectedVsp = v;
                break;
            }
        }
        if (selectedVsp == null) return;

        LocalDate selectedDate = datePicker.getValue();
        if (selectedDate == null) {
            showAlert("Выберите дату");
            return;
        }

        boolean useAlt = db.getFilialCheckName(currentUser.getFilialId());
        List<ChecklistTemplateItem> template = db.getChecklistTemplate();

        checklistContainer.getChildren().clear();

        Label header = new Label("Чек-лист для ВСП «" + selectedVsp.getName() +
                "» на " + selectedDate.format(DateTimeFormatter.ofPattern("dd.MM.yyyy")));
        header.setFont(Font.font("System", FontWeight.BOLD, 16));
        checklistContainer.getChildren().add(header);
        checklistContainer.getChildren().add(new Separator());

        for (ChecklistTemplateItem item : template) {
            HBox row = new HBox(10);
            row.setPadding(new Insets(8, 0, 8, 0));

            Label numLabel = new Label(String.valueOf(item.getOrder()));
            numLabel.setPrefWidth(30);
            Label descLabel = new Label(item.getDescription());
            descLabel.setWrapText(true);
            descLabel.setPrefWidth(400);
            Button detailsBtn = new Button("Подробнее");
            detailsBtn.setOnAction(e -> showItemDetails(item, useAlt, selectedVsp.getName(), detailsBtn));

            row.getChildren().addAll(numLabel, descLabel, detailsBtn);
            checklistContainer.getChildren().add(row);
            checklistContainer.getChildren().add(new Separator());
        }
    }

    private void showItemDetails(ChecklistTemplateItem item, boolean useAlt, String vspName, Button ownerButton) {
        String filterText = (useAlt && item.getAltFilterValue() != null && !item.getAltFilterValue().isEmpty())
                ? item.getAltFilterValue() : item.getFilterValue();
        String infoText = (useAlt && item.getAltAdditionalInfo() != null && !item.getAltAdditionalInfo().isEmpty())
                ? item.getAltAdditionalInfo() : item.getAdditionalInfo();
        String eventsText = (useAlt && item.getAltEventsValue() != null && !item.getAltEventsValue().isEmpty())
                ? item.getAltEventsValue() : item.getEventsValue();

        // Подстановка
        if (filterText != null) {
            filterText = filterText.replace("[Дата1]", LocalDate.now().format(DateTimeFormatter.ofPattern("dd.MM.yy")));
            filterText = filterText.replace("[РФ]", vspName);
        }

        PopOver popOver = new PopOver();
        TabPane tabPane = new TabPane();

        Tab filterTab = new Tab("Фильтр");
        VBox filterBox = new VBox(10);
        filterBox.setPadding(new Insets(10));

        Label infoLabel = new Label("Описание процедуры:");
        infoLabel.setFont(Font.font("System", FontWeight.BOLD, 12));
        Label infoTextLabel = new Label(infoText != null && !infoText.isEmpty() ? infoText : "Описание отсутствует");
        infoTextLabel.setWrapText(true);

        TextArea filterArea = new TextArea(filterText != null ? filterText : "Не задан");
        filterArea.setEditable(false);
        filterArea.setPrefRowCount(8);

        Button copyBtn = new Button("📋 Копировать фильтр");
        copyBtn.setStyle("-fx-background-color: #4CAF50; -fx-text-fill: white;");
        final String finalFilter = filterText != null ? filterText : "";
        copyBtn.setOnAction(e -> {
            ClipboardHelper.copyToClipboard(finalFilter);
            copyBtn.setText("✅ Скопировано!");
            new Thread(() -> {
                try { Thread.sleep(2000); } catch (InterruptedException ignored) {}
                Platform.runLater(() -> copyBtn.setText("📋 Копировать фильтр"));
            }).start();
        });

        filterBox.getChildren().addAll(infoLabel, infoTextLabel, new Label("Фильтр:"), filterArea, copyBtn);
        filterTab.setContent(filterBox);

        Tab eventsTab = new Tab("Мероприятия");
        TextArea eventsArea = new TextArea(eventsText != null ? eventsText : "Мероприятия не заданы");
        eventsArea.setEditable(false);
        eventsArea.setPrefRowCount(8);
        VBox eventsBox = new VBox(10, new Label("Мероприятия:"), eventsArea);
        eventsBox.setPadding(new Insets(10));
        eventsTab.setContent(eventsBox);

        tabPane.getTabs().addAll(filterTab, eventsTab);
        popOver.setContentNode(tabPane);
        popOver.setArrowLocation(PopOver.ArrowLocation.TOP_CENTER);
        popOver.show(ownerButton);
    }

    private void showAlert(String message) {
        Alert alert = new Alert(Alert.AlertType.WARNING, message, ButtonType.OK);
        alert.showAndWait();
    }
}
```

📌 Запуск (для Java 8)

1. Откройте терминал в папке проекта.
2. Выполните:
   ```bash
   mvn clean compile
   mvn javafx:run
   ```
   Плагин javafx-maven-plugin (com.zenjava) подхватит JavaFX из вашего JDK 8. Если используется OpenJDK 8 без JavaFX, раскомментируйте зависимости OpenJFX в pom.xml.

Проверка работы: появится окно авторизации. Введите логин пользователя, выберите ВСП и дату, нажмите «Показать чек-лист». Кнопка «Подробнее» открывает PopOver с фильтром и мероприятиями.

Проект полностью готов для JDK 1.8. Если возникнут ошибки времени выполнения, убедитесь, что у вас Oracle JDK 8 (включает JavaFX) или корректно подключены OpenJFX 8-зависимости.

    def get_filial_status_for_date(self, target_date):
        """
        Возвращает DataFrame со статусом каждого филиала на указанную дату.
        Статусы:
          - 'Заполнен'      – все ВСП филиала, кроме нерабочих, заполнили чек-лист.
          - 'Частично'      – хотя бы одно ВСП заполнило, но не все.
          - 'Не заполнен'   – ни одно активное ВСП не заполнило.
          - 'Все выходные'  – у всех ВСП филиала нерабочий день.
        """
        filials = self.get_filials()
        if filials.empty:
            return pd.DataFrame(columns=["Филиал", "Статус", "Всего ВСП", "Заполнено ВСП"])

        result = []
        for _, f in filials.iterrows():
            fid = int(f['id'])
            fname = f['name']

            # Все ВСП филиала
            vsp_list = self.get_vsp_by_filial(fid)

            # ВСП с нерабочим днём
            non_working_df = self._to_df(
                f"SELECT vsp_id FROM {self._table_name('vsp_non_working_days')} "
                f"WHERE date = %s AND filial_id = %s",
                (target_date, fid)
            )
            non_working_ids = set(non_working_df['vsp_id'].tolist()) if not non_working_df.empty else set()

            # ВСП с завершённой сессией
            completed_df = self._to_df(
                f"SELECT vsp_id FROM {self._table_name('checklist_sessions')} "
                f"WHERE operation_date = %s AND filial_id = %s AND status_bul = TRUE",
                (target_date, fid)
            )
            completed_ids = set(completed_df['vsp_id'].tolist()) if not completed_df.empty else set()

            total_vsp = len(vsp_list)
            active_vsp = total_vsp - len([v for v in vsp_list['id'] if v in non_working_ids])
            completed_vsp = len([v for v in vsp_list['id'] if v in completed_ids])

            # Определяем статус филиала
            if active_vsp == 0:
                status = "Все выходные"
            elif completed_vsp == active_vsp:
                status = "Заполнен"
            elif completed_vsp > 0:
                status = "Частично"
            else:
                status = "Не заполнен"

            result.append({
                "Филиал": fname,
                "Статус": status,
                "Всего ВСП": total_vsp,
                "Активных ВСП": active_vsp,
                "Заполнено ВСП": completed_vsp
            })

        return pd.DataFrame(result)
